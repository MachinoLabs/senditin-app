/** * SEND IT IN | Master Brain V14.2 * Thorough Build: Includes Sheet ID delivery for Dashboard Linking. */ function doGet(e) { const params = e.parameter; // 1. SECURITY KEY CHECK if (params.secretKey !== "MACHINO_V12_SECURE") { return ContentService.createTextOutput("DENIED: KEY").setMimeType(ContentService.MimeType.TEXT); } try { const masterSS = SpreadsheetApp.getActiveSpreadsheet(); const routingData = masterSS.getSheetByName("ROUTING_LOG").getDataRange().getValues(); let targetSheetId = ""; let authorized = false; const inputPIN = String(params.secretPIN || "").trim(); const inputCoID = String(params.companyID || "").trim().toUpperCase(); // 2. SEARCH FOR MATCH for (let i = 1; i < routingData.length; i++) { if (String(routingData[i][0]).trim() === inputPIN && String(routingData[i][1]).trim().toUpperCase() === inputCoID) { targetSheetId = routingData[i][2]; authorized = true; break; } } if (!authorized) return jsonResponse({error: "AUTH_FAILED"}); // --- MODE: READ (For Dashboard) --- if (params.mode === "read") { const clientSS = SpreadsheetApp.openById(targetSheetId); const rawSheet = clientSS.getSheetByName("RAW_DATA"); const data = rawSheet.getDataRange().getValues(); return jsonResponse({ headers: data[0].slice(4), entries: data.slice(1, 21), company: inputCoID, sheetId: targetSheetId }); } // --- MODE: CONFIG (For Field Toggles) --- if (params.mode === "config") { const clientSS = SpreadsheetApp.openById(targetSheetId); let configSheet = clientSS.getSheetByName("CONFIG") || clientSS.insertSheet("CONFIG"); if (params.saveData) { const settings = JSON.parse(params.saveData); configSheet.clear().appendRow(["Field", "Status"]); settings.forEach(s => configSheet.appendRow([s.name, s.status])); return ContentService.createTextOutput("SAVED"); } const configRows = configSheet.getDataRange().getValues(); // Auto-populate if empty if (configRows.length < 2) { configSheet.appendRow(["Field", "Status"]); configSheet.appendRow(["Assignment", "ON"]); return jsonResponse([["Field", "Status"], ["Assignment", "ON"]]); } return jsonResponse(configRows); } // --- MODE: WRITE (For Worker App) --- const timestamp = new Date(); let mapLink = params.lat && params.long ? "https://www.google.com/maps?q=" + params.lat + "," + params.long : "No GPS"; const fullRow = [[timestamp, inputCoID, inputPIN, params.workerName || "Unknown", mapLink, params.assignment || "", params.quantity || "", params.reference || "", params.notes || "", params.complete || ""]]; // Log to Master Backup masterSS.getSheetByName("ALL_REPORTS_BACKUP").insertRowBefore(2).getRange(2, 1, 1, 10).setValues(fullRow); // Log to Client Sheet const clientSS = SpreadsheetApp.openById(targetSheetId); clientSS.getSheetByName("RAW_DATA").insertRowBefore(2).getRange(2, 1, 1, 10).setValues(fullRow); return ContentService.createTextOutput("SUCCESS"); } catch (err) { return ContentService.createTextOutput("ERROR: " + err.toString()); } } function jsonResponse(obj) { return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); }
